%============================================================
% 现代中文 PhD 论文模板 - 宏包配置
% 使用 ctex 宏包 + XeLaTeX 编译
%============================================================

%---------- 基础宏包 ----------
% ctex: 中文支持核心宏包，自动处理字体和排版
% fontset=fandol: 使用 Fandol 字体（跨平台可用）
% fontset=windows: Windows 系统使用 SimSun/SimHei
% fontset=mac: macOS 系统使用 PingFang/Heiti
% 可根据系统选择或保持默认自动检测
\usepackage[UTF8,heading=true,zihao=-4,fontset=fandol]{ctex}

%---------- 页面布局 ----------
% geometry: 现代页面布局宏包
\usepackage{geometry}

%---------- 数学宏包 ----------
% amsmath: 基础数学排版
% amssymb: 数学符号
% amsfonts: 数学字体
% bm: 粗体数学符号
\usepackage{amsmath,amssymb,amsfonts}
\usepackage{bm}

%---------- 图表宏包 ----------
% graphicx: 插图支持
% booktabs: 专业表格线
% multirow: 表格多行合并
% array: 表格增强
% float: 浮动体控制
\usepackage{graphicx}
\usepackage{booktabs,multirow,array}
\usepackage{float}

%---------- 子图支持 ----------
% subcaption: 现代子图宏包（替代过时的 subfigure）
\usepackage{subcaption}

%---------- 标题格式 ----------
% caption: 图表标题格式（替代过时的 caption2）
\usepackage{caption}

%---------- 颜色与超链接 ----------
% xcolor: 颜色支持
% hyperref: 超链接和 PDF 书签
\usepackage{xcolor}
\usepackage[
    CJKbookmarks=true,
    unicode=true,
    bookmarksnumbered=true,
    bookmarksopen=true,
    colorlinks=false,
    citecolor=blue,
    linkcolor=red,
    anchorcolor=green,
    urlcolor=blue,
    pdfborder={0 0 0}
]{hyperref}

%---------- 参考文献 ----------
% natbib: 作者-年份和数字引用格式
% 保持与 JHEP 格式的兼容性
\usepackage[numbers,sort&compress,square]{natbib}

%---------- 页眉页脚 ----------
% fancyhdr: 页眉页脚控制
\usepackage{fancyhdr}

%---------- 目录格式 ----------
% titletoc: 目录格式控制（引导符、缩进等）
\usepackage{titletoc}

%---------- 其他实用宏包 ----------
% indentfirst: 首行缩进
\usepackage{indentfirst}

% enumitem: 列表环境增强
\usepackage{enumitem}

% setspace: 行距控制
\usepackage{setspace}

% listings: 代码排版（如需要）
% \usepackage{listings}

% algorithm2e: 算法排版（如需要）
% \usepackage[ruled,vlined]{algorithm2e}

%---------- 特殊字体 ----------
% mathrsfs: 花体数学字体
\usepackage{mathrsfs}

%---------- TikZ 绘图 ----------
\usepackage{tikz}
\usetikzlibrary{shapes,arrows,positioning,calc}

%---------- 脚注控制 ----------
% footmisc: 脚注格式
\usepackage[perpage,symbol*]{footmisc}

%---------- 日期时间 ----------
% datetime2: 现代日期时间宏包（替代 datetime）
\usepackage{datetime2}

%---------- 其他 ----------
% etoolbox: 编程工具
\usepackage{etoolbox}

% calc: 长度计算
\usepackage{calc}

% ifthen: 条件判断
\usepackage{ifthen}

%---------- 参考文献兼容设置 ----------
% 确保与 JHEP 格式的兼容性
\bibpunct{[}{]}{,}{n}{}{}

%---------- 自定义命令预定义 ----------
% 超引用命令
\newcommand{\supercite}[1]{\textsuperscript{\cite{#1}}}

% 量子力学符号
\newcommand{\ket}[1]{\left\vert{#1}\right\rangle}
\newcommand{\bra}[1]{\left\langle{#1}\right\vert}
\newcommand{\abs}[1]{\left\vert{#1}\right\vert}

%---------- 中文数字支持（向后兼容）----------
\newcommand{\CJKdigits}[1]{\zhdigits{#1}}
