%%   1. 定义文档类选项
%%   2. 加载所有宏包和格式设置
%%   3. 组织各章节文件的顺序
%%   4. 控制文档结构

%%   latexmk -xelatex main.tex

%%===========================================================================
%%  文档类设置
%%===========================================================================
%% 文档类: report - 适合撰写学位论文
%% 参数说明:
%%   - 12pt: 正文字号 12pt (小四号)
%%   - a4paper: A4 纸张大小
%%   - twoside: 双面打印模式(奇偶页不同)
%% 修改建议: 
%%   - 如需单面打印，改为 oneside
%%   - 如需更大字号，改为 14pt
\documentclass[12pt,a4paper,twoside]{report}

%%===========================================================================
%%  加载宏包配置
%%===========================================================================
%% 加载 setup/package.tex 中的宏包配置
%% 包含所有必要的 LaTeX 宏包
\input{setup/package.tex}

%%===========================================================================
%%  图片路径设置
%%===========================================================================
%% 设置图片搜索路径
%% 搜索顺序: chapter/ -> figures/
%% 使用: \includegraphics{filename} 即可自动搜索
\graphicspath{{chapter/}{figures/}}

%%===========================================================================
%%  加载格式设置
%%===========================================================================
%% 加载 setup/format.tex 中的格式配置
%% 包含页面布局、页眉页脚、章节格式等
\input{setup/format.tex}

%%===========================================================================
%%  文档开始
%%===========================================================================
\begin{document}

%%--------------------------------------------------------------------------
%%  封面部分 (Roman 页码前)
%%--------------------------------------------------------------------------
%% 中文封面
%% 用途: 显示论文标题、作者、导师、学位等信息
\include{chapter/A_Title_C}

%% 英文封面
%% 用途: 显示英文论文信息
\include{chapter/A_Title_E}

%% 原创性声明和版权授权书
%% 用途: 学校要求的学位论文声明页
\include{chapter/B_originauth}

%%--------------------------------------------------------------------------
%%  摘要部分 (Roman 页码)
%%--------------------------------------------------------------------------

%% 切换页码为罗马数字 (I, II, III...)
\pagenumbering{Roman}

%% 中文摘要
%% 用途: 概述论文研究内容、方法、结果和结论
%% 要求: 一般 500-1000 字，关键词 3-5 个
\include{chapter/C_Abstract_C}

%% 英文摘要
%% 用途: 中文摘要的英文翻译
%% 要求: 与中文摘要内容对应
\include{chapter/C_Abstract_E}

%%--------------------------------------------------------------------------
%%  目录
%%--------------------------------------------------------------------------
%% 自动生成目录
%% 包含: 章节标题、图表目录、参考文献等
\clearpage
\pagestyle{empty}
\tableofcontents
\clearpage

%%--------------------------------------------------------------------------
%%  正文部分 (阿拉伯数字页码)
%%--------------------------------------------------------------------------
%% 切换页码为阿拉伯数字 (1, 2, 3...)
%% 从 1 开始计数
\pagenumbering{arabic}
\setcounter{page}{1}
\fancyfoot[C]{\thepage}

%% 第一章: 引言
%% 用途: 介绍研究背景、意义、现状和论文结构
\include{chapter/D_introduction}


%% 第二章: 主体内容
%% 用途: 理论基础、研究方法等
\include{chapter/E_chapter_two}

%% 第三章: 主体内容
%% 用途: 实验设计、数据分析等
\include{chapter/F_chapter_three}

%% 第四章: 主体内容
%% 用途: 结果讨论、对比分析等
\include{chapter/G_chapter_four}

%% 第五章: 总结与展望
%% 用途: 总结研究成果，提出未来研究方向
\include{chapter/H_conclusions}

%%--------------------------------------------------------------------------
%%  附录部分
%%--------------------------------------------------------------------------
%% 附录 A
%% 用途: 补充材料、详细推导、原始数据等
\setappendixA
\include{chapter/I_appendixa}

%% 附录 B
\setappendixB
\include{chapter/J_appendixb}

%%--------------------------------------------------------------------------
%%  参考文献
%%--------------------------------------------------------------------------
%% 设置参考文献样式
%% JHEP: 高能物理领域常用样式
\bibliographystyle{JHEP}

%% 加载参考文献数据库
%% Refs.bib 包含所有引用条目
\bibliography{Refs}

%%--------------------------------------------------------------------------
%%  发表论文列表
%%--------------------------------------------------------------------------
%% 用途: 列出攻读学位期间发表的学术论文
\include{chapter/O_papers}

%%--------------------------------------------------------------------------
%%  致谢
%%--------------------------------------------------------------------------
%% 用途: 感谢导师、同学、家人等
%% 注意: 此部分应真挚、简洁
\include{chapter/P_thanks}

%%===========================================================================
%%  文档结束
%%===========================================================================
\end{document}
